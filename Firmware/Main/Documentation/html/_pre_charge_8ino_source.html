70void preChargeCircuitFSMTransitions (PreChargeTaskData preChargeData) {
71 HV_STATE old_state = hv_state;
72 switch (hv_state) { // transitions
73 case HV_OFF:
74 if (check_HV_TOGGLE()) {
75 hv_state = HV_PRECHARGING;
76 }
77 break;
78 case HV_PRECHARGING:
79 if (!check_HV_TOGGLE()) {
80 // kill-switch activated or HV switch turned off
81 hv_state = HV_OFF;
82 }
83 else if (!isHVSafe(preChargeData)) {
84 // HV error detected
85 hv_state = HV_ERROR;
86 }
87 else if (isPrecharged(preChargeData)) {
88 // finished precharging
89 hv_state = HV_ON;
90 }
91 else {
92 // no updates, keep precharging
93 hv_state = HV_PRECHARGING;
94 }
95 break;
96 case HV_ON:
97 // TODO: add another && next to the !check_HV_TOGGLE, that essentially checks the measured angle > 45 degrees on left or right side (+- 45 degrees?)
98 if (!check_HV_TOGGLE() || *preChargeData.angle_Y > 45 || *preChargeData.angle_Y < -45 || *preChargeData.angle_X > 45 || *preChargeData.angle_X < -45) {
99 // kill-switch activated or HV switch turned off
100 hv_state = HV_OFF;
101 }
102 else if (!isHVSafe(preChargeData)) {
103 // HV error detected
104 hv_state = HV_ERROR;
105 }
106 else {
107 // no updates, keep HV on
108 hv_state = HV_ON;
109 }
110 break;
111 case HV_ERROR:
112 if (isHVSafe(preChargeData)) {
113 // if the error has been cleared
114 hv_state = HV_OFF;
115 } else {
116 // otherwise stay here
117 hv_state = HV_ERROR;
118 }
119 break;
120 default:
121 hv_state = HV_OFF;
122 break;
123 } // transitions
124
125 if (hv_state != old_state) {
126 Serial.printf("HV transitioned from %s to %s state\n", state_name(old_state), state_name(hv_state));
127 }
128}

203void setupI2C(PreChargeTaskData preChargeData) {
204 Wire.setClock(400000); // MPU6050 supports up to 400k Hz in specifications
205 Wire.begin();
206 delay(50); // give delay for device to start
207
208 // Start gyro in power mode
209 Wire.beginTransmission(0x68);
210 Wire.write(0x6B); // 6B is relevant register
211 Wire.write(0x00); // all bits must be 0 to start and continue device
212 Wire.endTransmission();
213
214 // Perform gyroscope calibration measurements
215 // 2000 milliseconds = 2 seconds to add all measured variables to calibration variables
216 // This is important because this solves the issue of a non-zero rotation rate when stationary
217 for (*preChargeData.RateCalibrationNumber = 0; *preChargeData.RateCalibrationNumber < 2000; *preChargeData.RateCalibrationNumber++) {
218 gyro_signals(preChargeData);
219 *preChargeData.RateCalibrationRoll += *preChargeData.RateRoll;
220 *preChargeData.RateCalibrationPitch += *preChargeData.RatePitch;
221 *preChargeData.RateCalibrationYaw += *preChargeData.RateYaw;
222 delay(1);
223 }
224
225 // Take average of calibrated rotation rate values from each direction
226 *preChargeData.RateCalibrationRoll /= 2000;
227 *preChargeData.RateCalibrationPitch /= 2000;
228 *preChargeData.RateCalibrationYaw /= 2000;
229 // *preChargeData.LoopTimer = micros();
230}

232void gyro_signals(PreChargeTaskData preChargeData) {
233 // Start I2C communication with MPU6050
234 Wire.beginTransmission(0x68); // 0x68 is default register value for MPU6050
235
236 // Switch on low pass filter
237 Wire.write(0x1A); // activate low pass filter
238 Wire.write(0x05); // cut off frequency of 10 Hz
239 Wire.endTransmission();
240
241 // Configure accelerometer output
242 Wire.beginTransmission(0x68);
243 Wire.write(0x1C); // 1C is relevant register
244 Wire.write(0x10); // full scale range of +/-8g
245 Wire.endTransmission();
246
247 // Access registers storing accelerometer measurements
248 Wire.beginTransmission(0x68);
249 Wire.write(0x3B);
250 Wire.endTransmission();
251 Wire.requestFrom(0x68, 6);
252
253 // Read accelerometer measurements
254 int16_t AccXLSB = Wire.read() << 8 | Wire.read(); // x-direction
255 int16_t AccYLSB = Wire.read() << 8 | Wire.read(); // y-direction
256 int16_t AccZLSB = Wire.read() << 8 | Wire.read(); // z-direction
257
258 // Configure gyroscope output and pull rotation rate measurements from sensor
259 // Set sensitivity scale factor
260 Wire.beginTransmission(0x68);
261 Wire.write(0x1B); // 1B is hexadecimal associated with gyroscope configuration
262 Wire.write(0x8); // 8 is hexadecimal for LSB sensitivity of 65.6 LSB/degree/second
263 Wire.endTransmission();
264
265 // Access registers storing gyro measurements
266 Wire.beginTransmission(0x68);
267 Wire.write(0x43);
268 Wire.endTransmission();
269 Wire.requestFrom(0x68, 6);
270
271 // Read gyro measurements
272 int16_t GyroX = Wire.read() << 8 | Wire.read(); // around x-axis
273 int16_t GyroY = Wire.read() << 8 | Wire.read(); // around y-axis
274 int16_t GyroZ = Wire.read() << 8 | Wire.read(); // around z-axis
275
276 // Convert measurement units to degree/second
277 *preChargeData.RateRoll = (float)GyroX / 65.5;
278 *preChargeData.RatePitch = (float)GyroY / 65.5;
279 *preChargeData.RateYaw = (float)GyroZ / 65.5;
280
281 // Convert measurements to from LSB to g
282 // Divide 4096 because full range +/-8g is associated with 4096 LSB/g
283 *preChargeData.AccX = (float)AccXLSB / 4096;
284 *preChargeData.AccY = (float)AccYLSB / 4096;
285 *preChargeData.AccZ = (float)AccZLSB / 4096;
286
287 float AccX = *preChargeData.AccX;
288 float AccY = *preChargeData.AccY;
289 float AccZ = *preChargeData.AccZ;
290
291 // Calculate absolute angles
292 // IMPORTANT LINE OF CODE:
293 // - WE CAN ADD OR SUBTRACT FROM THE ANGLE DEPENDING ON THE PLACEMENT OF GYROSCOPE
294 // - EX: If MPU6050 is laid on port side, add 90 to AnglePitch
295 *preChargeData.AngleRoll = (atan(AccY / sqrt(AccX * AccX + AccZ * AccZ)) * 1 / (3.142 / 180)) + 5;
296 *preChargeData.AnglePitch = (-atan(AccX / sqrt(AccY * AccY + AccZ * AccZ)) * 1 / (3.142 / 180));
297}

299void updateGyroData(PreChargeTaskData preChargeData) {
300 gyro_signals(preChargeData);
301
302 *preChargeData.RateRoll -= *preChargeData.RateCalibrationRoll;
303 *preChargeData.RatePitch -= *preChargeData.RateCalibrationPitch;
304 *preChargeData.RateYaw -= *preChargeData.RateCalibrationYaw;
305
306 // Calculate Roll angle (around x axis)
307 kalman_1d(*preChargeData.angle_X, *preChargeData.KalmanUncertaintyAngleRoll, *preChargeData.RateRoll, *preChargeData.AngleRoll, preChargeData);
308
309 // Update Kalman output to angle roll and uncertaintity
310 *preChargeData.angle_X = preChargeData.Kalman1DOutput[0];
311 *preChargeData.KalmanUncertaintyAngleRoll = preChargeData.Kalman1DOutput[1];
312

318 *preChargeData.KalmanUncertaintyAnglePitch = preChargeData.Kalman1DOutput[1];

321void kalman_1d(float KalmanState, float KalmanUncertainty, float KalmanInput, float KalmanMeasurement, PreChargeTaskData preChargeData) {
322 KalmanState = KalmanState + 0.004 * KalmanInput;
323 KalmanUncertainty = KalmanUncertainty + 0.004 * 0.004 * 4 * 4;
324 float KalmanGain = KalmanUncertainty * 1 / (1 * KalmanUncertainty + 3 * 3);
325 KalmanState = KalmanState + KalmanGain * (KalmanMeasurement - KalmanState);
326 KalmanUncertainty = (1 - KalmanGain) * KalmanUncertainty;
327 // Output of filter
328 preChargeData.Kalman1DOutput[0] = KalmanState;
329 preChargeData.Kalman1DOutput[1] = KalmanUncertainty;
330}

130void thermiDataPrint(int numberOfLines) {
131 // number of thermistor values to print per line
132 int incr = NUM_THERMI / numberOfLines;
133 for (int i = 0; i < numberOfLines; i++) {
134 bool sameData = true;
135 String sOld, sNew;
136 // process data/update display one line of therm values at a time
137 for (int j = i * incr; j < incr + (i * incr); j++) {
138 if (thermiData.currData[j] != thermiData.oldData[j]) {
139 sameData = false;
140 sNew.append((byte)thermiData.currData[j]);
141 sOld.append((byte)thermiData.oldData[j]);
142 thermiData.oldData[j] = thermiData.currData[j];
143 } else {
144 sOld.append((byte)thermiData.oldData[j]);
145 sNew.append((byte)thermiData.oldData[j]);
146 }
147 // if not the last value on a line, add a comma
148 if (j != (incr + (i * incr)) - 1) {
149 sOld.append(", ");
150 sNew.append(", ");
151 }
152 }
153 // if therm data has changed since the last print, update the display
154 if (!sameData) {
155 eraseThenPrint(thermiData.dataX, thermiData.y + (VERTICAL_SCALER * i), sOld, sNew);
156 }
157 }

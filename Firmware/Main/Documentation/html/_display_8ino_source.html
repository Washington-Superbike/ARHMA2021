104 eraseThenPrint(printedVals[i].dataX, printedVals[i].y, printedVals[i].oldData, *printedVals[i].currData);

117 if (screen.screenType == SPEEDOMETER){
118 tft.setTextColor(ILI9341_WHITE);
119 tft.setTextSize(8);
120 String newString = "";
121 String oldString = "";
122 if ((*printedVals[3].currData) != (printedVals[3].oldData) || (printedVals[3].oldData) == DEFAULT_FLOAT) {
123// Gear ratio, 48 teeth in the back wheel sprocoket. 16 on motor sprocket
124// Diameter = 0.522 m, divided by 60 converts it into per second, so the RPM is converted to a final
125// Speed of m/s
126 float gearRatio = 48/16;
127 float diameterPerSecond = 0.522/60;
128 float pi = 3.14159;
129 float mphConvert = 2.2369362920544;
130
131 float newSpeed = *printedVals[3].currData / gearRatio * pi * diameterPerSecond * mphConvert;
132 float oldSpeed = printedVals[3].oldData / gearRatio * pi * diameterPerSecond * mphConvert;
133 newString = (String) (int) newSpeed;
134 oldString = (String) (int) oldSpeed;
135// Serial.println(newString);
136// Serial.println(oldString);
137 eraseThenPrintSPEEDO(175, 0, oldString, newString);
138 newString = (String) (int) *printedVals[3].currData;
139 oldString = (String) (int) printedVals[3].oldData;
140 eraseThenPrintSPEEDO(120, 180, oldString, newString);
141 printedVals[3].oldData = *printedVals[3].currData;
142 }
143 }

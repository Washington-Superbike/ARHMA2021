32//dataLoggingTask processes all of the data logs and formats each CSV file output
33void dataLoggingTask(void *dlData) {
34 DataLoggingTaskData dl = *(DataLoggingTaskData *)dlData;
35 int sTime;

38 int mTime = millis();
39 sTime = (millis() - epochTime) / 1000;
40 for (int i = 0; i < dl.writersLen; i++) {
41 addRecord(dl.writers[i], sTime);
42 }
43 if ((millis() - lastSave) > 10000) {
44 Serial.print("Saving files...");
45 saveFiles(dl.writers, dl.writersLen);
46 Serial.println("saved");
47 lastSave = millis();
48 }
49 // delay 50ms (change to modify how fast records are saved

51 // ex: motor current should write every 5ms but battery voltage only every 1 second

54 }
55}

33void canTask(void *canData) {
34 int iter = 0;
35 int requestCells = 1;
36 while (1) {
37 // check for new incoming messages
38
39 // CAN breaks if we try sending messages with 0 other nodes on the bus
40 // as there is no node to 'ACK' our message. Therefore,
41 // change CAN_NODES in Main.h to make sure things dont break.
42 if (CAN_NODES !=0) {
43 checkCAN(*(CANTaskData *)canData);
44 if (iter == (1000 / 20) * 2) {
45 // ask for other half of cell voltages from BMS every 2 seconds
46 requestCellVoltages(requestCells);
47 requestCells *= -1;
48 iter = 0;
49 }
50 iter++;
51 }
52 // delay 20ms
53 vTaskDelay((20 * configTICK_RATE_HZ) / 1000);
54 }
55}

146// checks the can bus for any new data
147void checkCAN(CANTaskData canData) {
148 int readValue = CAN_bus.read(CAN_msg);
149 if (readValue != 0) { // if we read a message
150 switch (CAN_msg.id) {
151 case MOTOR_STATS_MSG:
152 decodeMotorStats(CAN_msg, canData.motorStats);
153 break;
154 case MOTOR_TEMPS_MSG:
155 decodeMotorTemps(CAN_msg, canData.motorTemps);
156 break;
157 case DD_BMS_STATUS_IND:
158 decipherBMSStatus(CAN_msg, canData.bmsStatus);
159 //printBMSStatus();
160 break;
161 case EVCC_STATS:
162 decipherEVCCStats(CAN_msg, canData.chargeControllerStats);
163 break;
164 case CHARGER_STATS:
165 decipherChargerStats(CAN_msg, canData.chargerStats);
166 case BMSC1_LTC1_CELLS_04:
167 decipherCellsVoltage(CAN_msg, canData.cellVoltages);
168 break;
169 case BMSC1_LTC1_CELLS_58:
170 decipherCellsVoltage(CAN_msg, canData.cellVoltages);
171 break;
172 case BMSC1_LTC1_CELLS_912:
173 decipherCellsVoltage(CAN_msg, canData.cellVoltages);
174 break;
175 case BMSC1_LTC2_CELLS_04:
176 decipherCellsVoltage(CAN_msg, canData.cellVoltages);
177 break;
178 case BMSC1_LTC2_CELLS_58:
179 decipherCellsVoltage(CAN_msg, canData.cellVoltages);
180 break;
181 case BMSC1_LTC2_CELLS_912:

183 break;

185 decipherThermistors(CAN_msg, canData.thermistorTemps);

187 }
